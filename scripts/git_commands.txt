# Git commands to create branch, commit, push, and open PR (PowerShell)

git checkout -b ci/cloudflare-pages
git add .
git commit -m "chore(deploy): switch CI to Cloudflare Pages and add deploy workflow"
git remote add origin <repo-url>  # only if remote not set
git push -u origin ci/cloudflare-pages
